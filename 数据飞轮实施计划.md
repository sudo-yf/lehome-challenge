    # 1. 轨迹长度合理 (50-600 frames)
    if len(episode_data) < 50 or len(episode_data) > 600:
        return False

    # 2. 动作平滑度检查
    actions = np.array([frame["action"] for frame in episode_data])

### 阶段3: 数据集合并 (1小时)

#### 3.1 合并脚本
```python
# scripts/merge_datasets.py
from lerobot.datasets.lerobot_dataset import LeRobotDataset
from pathlib import Path

def merge_datasets(base_path, eval_path, output_path):
    """合并原始数据集和评估收集的数据"""

    # 加载数据集
    base_ds = LeRobotDataset(base_path)
    eval_ds = LeRobotDataset(eval_path)

    # 创建新数据集
    merged_ds = LeRobotDataset.create(
        repo_id="top_long_merged_v2",
        fps=30,
        root=output_path,
        use_videos=True,
        features=dict(base_ds.meta.features),
    )

    # 复制原始数据
    print(f"复制原始数据集: {base_ds.num_episodes} episodes")
    for ep_idx in range(base_ds.num_episodes):
        frames = base_ds.get_episode(ep_idx)
        for frame in frames:
            merged_ds.add_frame(frame)
        merged_ds.save_episode()

    # 添加评估数据
    print(f"添加评估数据: {eval_ds.num_episodes} episodes")
    added_count = 0
    for ep_idx in range(eval_ds.num_episodes):
        frames = eval_ds.get_episode(ep_idx)
        if should_add_episode(frames):
            for frame in frames:
                merged_ds.add_frame(frame)
            merged_ds.save_episode()
            added_count += 1

    merged_ds.finalize()
    print(f"合并完成: {base_ds.num_episodes} + {added_count} = {merged_ds.num_episodes} episodes")
    return merged_ds

# 执行合并
merge_datasets(
    base_path="Datasets/example/top_long_merged",
    eval_path="Datasets/eval_collected/最新目录",
    output_path="Datasets/example/top_long_merged_v2"
)
```

## ⚠️ 注意事项

### 1. 数据质量控制
- ✅ 只保存成功的episode
- ✅ 检查轨迹平滑度
- ✅ 避免过度相似的轨迹
- ⚠️ 定期人工抽查数据质量

### 2. 计算资源
- 每轮训练: ~8-10小时 (GPU)
- 每轮评估: ~30分钟
- 数据合并: ~10分钟
- 总周期: ~1天/轮

### 3. 存储空间
- 原始数据集: ~10GB
- 每轮新增: ~1-2GB
- 建议预留: 50GB

### 4. 过拟合风险
- 监控训练/验证loss
- 如果新数据过少，可能导致过拟合
- 建议每轮至少添加15-20个新episode

## 🎓 学习和改进

### 数据飞轮的优势
1. **自动化**: 无需人工标注，模型自己生成训练数据
2. **持续改进**: 每轮迭代都能提升性能
3. **探索新策略**: 模型可能发现新的成功路径
4. **成本效益**: 利用现有评估流程，额外成本低

### 潜在问题和解决方案
| 问题 | 解决方案 |
|------|---------|
| 数据分布偏移 | 定期混合原始数据和新数据 |
| 过拟合 | 增加数据多样性，使用数据增强 |
| 性能平台期 | 调整环境难度，引入新场景 |
| 计算资源不足 | 优化训练参数，使用更小的batch size |
